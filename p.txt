// FILE BARU: Tambahkan ini sebagai whale-analyzer-pro.ts

export interface WhaleQualityScore {
  totalScore: number;
  sharpeRatio: number;
  avgEntryTiming: number;
  realizedPnL_7d: number;
  copyCatRatio: number;
  mevInteraction: boolean;
  rugHistory: boolean;
}

export class ProWhaleAnalyzer {
  private whaleHistory = new Map<string, any>();
  
  async analyzeWhale(whaleAddress: string): Promise<WhaleQualityScore> {
    // Ambil 30 transaction terakhir
    const txs = await this.getWhaleTransactions(whaleAddress, 30);
    
    // Filter hanya transaksi Uniswap
    const swaps = txs.filter(tx => tx.method === 'swap');
    
    // Hitung realized profit 7 hari
    const last7DaysProfit = this.calculate7DayProfit(swaps);
    
    // Hitung timing entry (beli di awal pump?)
    const entryTiming = this.calculateEntryTiming(swaps);
    
    // Deteksi MEV (sandwich attack)
    const mevDetected = this.detectMEV(swaps);
    
    // Hitung total score
    const score = this.calculateScore({
      sharpeRatio: this.calculateSharpeRatio(swaps),
      avgEntryTiming: entryTiming,
      realizedPnL_7d: last7DaysProfit,
      copyCatRatio: await this.getCopyCatRatio(whaleAddress),
      mevInteraction: mevDetected,
      rugHistory: await this.checkRugHistory(whaleAddress)
    });
    
    return score;
  }
  
  async crowdCopySignal(whaleAddresses: string[], token: string): Promise<number> {
    let totalScore = 0;
    for (const whale of whaleAddresses) {
      const isBuying = await this.isWhaleBuyingToken(whale, token);
      if (isBuying) {
        const score = await this.getWhaleScore(whale);
        totalScore += score;
      }
    }
    return totalScore; // Threshold: 350 untuk execute
  }
  
  private calculateSharpeRatio(txs: any[]): number {
    const returns = txs.map(tx => (tx.profitPercent || 0) / 100);
    const avgReturn = returns.reduce((a,b) => a+b, 0) / returns.length;
    const variance = returns.map(r => Math.pow(r - avgReturn, 2)).reduce((a,b) => a+b, 0) / returns.length;
    const stdDev = Math.sqrt(variance);
    return stdDev === 0 ? 0 : avgReturn / stdDev;
  }
  
  private calculateEntryTiming(txs: any[]): number {
    // Logic: beli di 20% pertama pump = 100 score
    let totalTiming = 0;
    for (const tx of txs) {
      const pumpStage = await this.getPumpStage(tx.token, tx.blockNumber);
      totalTiming += Math.max(0, 100 - (pumpStage * 100));
    }
    return totalTiming / txs.length;
  }
  
  private async getCopyCatRatio(whaleAddress: string): Promise<number> {
    // Berapa banyak trader lain mengikuti whale ini
    const followers = await this.getCopyTraders(whaleAddress);
    return Math.min(followers.length / 100, 1);
  }
  
  private async getPumpStage(token: string, blockNumber: number): Promise<number> {
    // 0 = awal pump, 1 = akhir pump
    const currentPrice = await this.getPrice(token);
    const initialBlocks = await this.getFirstBlocks(token);
    const blocksPassed = blockNumber - initialBlocks;
    return Math.min(blocksPassed / 30, 1); // 30 block = pump selesai
  }
}

export class MempoolAnalyzer {
  private pendingTransactions: Map<string, any> = new Map();
  
  async detectWhalePending(whaleAddress: string): Promise<any | null> {
    const pendingTxs = await this.getPendingTransactions();
    const whaleTx = pendingTxs.find(tx => 
      tx.from.toLowerCase() === whaleAddress.toLowerCase() &&
      (tx.to === UNISWAP_ROUTER || tx.data.includes('0x414bf389'))
    );
    
    if (whaleTx) {
      return {
        token: this.extractTokenFromData(whaleTx.data),
        amount: whaleTx.value,
        gasPrice: whaleTx.gasPrice,
        blockNumber: whaleTx.blockNumber
      };
    }
    return null;
  }
  
  async executeFrontrun(whaleTx: any, myCapital: number): Promise<boolean> {
    // Eksekusi dengan gas price 20% lebih tinggi
    const myGasPrice = BigInt(whaleTx.gasPrice) * BigInt(120) / BigInt(100);
    
    const tx = {
      to: UNISWAP_ROUTER,
      data: this.buildSwapData(whaleTx.token, myCapital),
      gasPrice: myGasPrice,
      gasLimit: 300000
    };
    
    return await this.sendTransaction(tx);
  }
  
  async waitForPullback(token: string, whaleBuyPrice: number): Promise<boolean> {
    // Tunggu harga turun 3-5% dari harga beli whale
    return new Promise((resolve) => {
      const interval = setInterval(async () => {
        const currentPrice = await this.getTokenPrice(token);
        const dropPercent = (whaleBuyPrice - currentPrice) / whaleBuyPrice;
        
        if (dropPercent >= 0.03) { // 3% drop
          clearInterval(interval);
          resolve(true);
        }
      }, 1000);
      
      // Timeout 2 menit
      setTimeout(() => {
        clearInterval(interval);
        resolve(false);
      }, 120000);
    });
  }
  
  private async getPendingTransactions() {
    // Gunakan RPC yang support pending transactions
    const provider = new ethers.WebSocketProvider('wss://base.llamarpc.com');
    const pending = await provider.send('eth_getBlockByNumber', ['pending', true]);
    return pending.transactions;
  }
}

export interface Position {
  token: string;
  entryPrice: number;
  currentPrice: number;
  profitPercent: number;
  entryTime: Date;
}

export class DynamicExitStrategy {
  async calculateExit(position: Position): Promise<string> {
    const momentum1m = await this.getPriceMomentum(position.token, 60);
    const volumeMomentum = await this.getVolumeMomentum(position.token, 60);
    
    // PROFIT BESAR >30%
    if (position.profitPercent > 30) {
      // Panic sell if price drops >5% in 1 minute
      if (momentum1m < -5) {
        return "SELL_ALL_PANIC";
      }
      // Exit half if price drops >2% AND volume drops >30%
      if (momentum1m < -2 && volumeMomentum < -30) {
        return "SELL_50_PERCENT";
      }
    }
    
    // SCALING OUT (ambil profit bertahap)
    if (position.profitPercent >= 40) {
      return "SELL_25_PERCENT";
    }
    if (position.profitPercent >= 80) {
      return "SELL_50_PERCENT";
    }
    if (position.profitPercent >= 150) {
      return "SELL_ALL";
    }
    
    // DYNAMIC STOP LOSS (tighten jika profit)
    let trailingPercent = 15; // default
    if (position.profitPercent > 20) {
      trailingPercent = 8;
    } else if (position.profitPercent > 0) {
      trailingPercent = 12;
    }
    
    // Check trailing stop condition
    const highestPrice = await this.getHighestPriceSinceEntry(position.token, position.entryPrice);
    const peakDrop = (highestPrice - position.currentPrice) / highestPrice * 100;
    
    if (peakDrop >= trailingPercent) {
      return "SELL_ALL_TRAILING";
    }
    
    // Waktu maksimal hold 15 menit
    const holdMinutes = (Date.now() - position.entryTime.getTime()) / 60000;
    if (holdMinutes >= 15 && position.profitPercent > 0) {
      return "SELL_ALL_TIMEOUT";
    }
    
    return "HOLD";
  }
  
  private async getPriceMomentum(token: string, seconds: number): Promise<number> {
    const priceNow = await this.getPrice(token);
    const priceSecondsAgo = await this.getPriceAtTime(token, Date.now() - seconds * 1000);
    return ((priceNow - priceSecondsAgo) / priceSecondsAgo) * 100;
  }
  
  private async getVolumeMomentum(token: string, seconds: number): Promise<number> {
    const volumeNow = await this.getVolumeLastMinute(token);
    const volumePrev = await this.getVolumeBefore(token, seconds);
    return ((volumeNow - volumePrev) / volumePrev) * 100;
  }
  
  private async getHighestPriceSinceEntry(token: string, entryPrice: number): Promise<number> {
    // Ambil highest price sejak entry dari GeckoTerminal
    const candles = await this.getOHLCV(token, 20);
    const highest = Math.max(...candles.map(c => c.high), entryPrice);
    return highest;
  }
}

export class MicroCapRiskManager {
  private todayLoss: number = 0;
  private consecutiveLosses: number = 0;
  private lastTradeProfit: number = 0;
  private cooldownUntil: Date = new Date();
  private totalCapital: number;
  private readonly INITIAL_CAPITAL = 0.006; // 0.006 ETH
  
  constructor(initialCapital: number = 0.006) {
    this.totalCapital = initialCapital;
  }
  
  async beforeTrade(token: any): Promise<{allowed: boolean, reason?: string}> {
    // Check cooldown
    if (this.cooldownUntil > new Date()) {
      const remaining = Math.ceil((this.cooldownUntil.getTime() - Date.now()) / 60000);
      return {allowed: false, reason: `Cooldown: ${remaining} menit tersisa`};
    }
    
    // Check daily loss limit
    if (this.todayLoss >= 0.0015) { // 25% dari 0.006
      return {allowed: false, reason: `Daily loss limit reached: ${this.todayLoss.toFixed(4)} ETH`};
    }
    
    // Check consecutive losses
    if (this.consecutiveLosses >= 3) {
      this.cooldownUntil = new Date(Date.now() + 30 * 60000);
      return {allowed: false, reason: `3 consecutive losses, cooldown 30 minutes`};
    }
    
    // Check minimum liquidity
    if (token.liquidityUSD < 5000) {
      return {allowed: false, reason: `Low liquidity: $${token.liquidityUSD}`};
    }
    
    // Check token age
    const ageSeconds = (Date.now() - token.createdAt) / 1000;
    if (ageSeconds > 300) { // 5 menit
      return {allowed: false, reason: `Token too old: ${Math.floor(ageSeconds)} detik`};
    }
    
    return {allowed: true};
  }
  
  afterTrade(profitETH: number) {
    if (profitETH < 0) {
      // Rugi
      this.todayLoss += Math.abs(profitETH);
      this.consecutiveLosses++;
      this.lastTradeProfit = profitETH;
    } else {
      // Profit
      this.consecutiveLosses = 0;
      this.lastTradeProfit = profitETH;
      
      // Cooldown setelah profit besar (>50% modal)
      if (profitETH > 0.003) {
        this.cooldownUntil = new Date(Date.now() + 15 * 60000);
      }
      
      // REINVEST ONLY 50% PROFIT
      const reinvestAmount = profitETH * 0.5;
      const withdrawAmount = profitETH * 0.5;
      this.totalCapital += reinvestAmount;
      
      console.log(`Profit: ${profitETH.toFixed(6)} ETH | Reinvest: ${reinvestAmount.toFixed(6)} | Withdraw: ${withdrawAmount.toFixed(6)}`);
    }
  }
  
  getPositionSize(): number {
    // Dynamic position sizing: 12% dari modal
    const baseSize = this.totalCapital * 0.12;
    // Clamp antara 0.001 ETH dan 0.003 ETH
    return Math.min(Math.max(baseSize, 0.001), 0.003);
  }
  
  resetDaily() {
    const now = new Date();
    if (now.getHours() === 0 && now.getMinutes() === 0) {
      this.todayLoss = 0;
    }
  }
}

// FILE BARU: Optimasi performa untuk modal kecil

export class PerformanceOptimizer {
  // Cache untuk mengurangi API calls
  private priceCache = new Map<string, {price: number, timestamp: number}>();
  private readonly CACHE_TTL = 3000; // 3 detik
  
  async getCachedPrice(token: string): Promise<number> {
    const cached = this.priceCache.get(token);
    if (cached && Date.now() - cached.timestamp < this.CACHE_TTL) {
      return cached.price;
    }
    
    const price = await this.fetchPriceFromGecko(token);
    this.priceCache.set(token, {price, timestamp: Date.now()});
    return price;
  }
  
  // Batch API calls
  async batchGetPrices(tokens: string[]): Promise<Map<string, number>> {
    const results = new Map();
    const uncached = tokens.filter(t => !this.priceCache.has(t));
    
    if (uncached.length > 0) {
      const prices = await this.fetchBatchPrices(uncached);
      uncached.forEach(t => {
        const price = prices[t];
        this.priceCache.set(t, {price, timestamp: Date.now()});
        results.set(t, price);
      });
    }
    
    tokens.forEach(t => {
      if (!results.has(t)) {
        results.set(t, this.priceCache.get(t)!.price);
      }
    });
    
    return results;
  }
  
  // Prioritaskan wallet yang paling profitable
  async prioritizeWallets(wallets: string[]): Promise<string[]> {
    const scores = await Promise.all(
      wallets.map(async w => ({
        wallet: w,
        score: await this.getWhaleProfitability(w)
      }))
    );
    
    return scores
      .sort((a,b) => b.score - a.score)
      .slice(0, 5)
      .map(s => s.wallet);
  }
}
